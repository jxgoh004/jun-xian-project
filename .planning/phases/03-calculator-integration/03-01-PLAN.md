**5. Add JavaScript before closing `</body>` tag (per D-03, D-04, D-05, D-06):**

```html
<script>
(function() {
  const homeView = document.getElementById('home-view');
  const projectView = document.getElementById('project-view');

  const projects = {
    calculator: 'projects/calculator/index.html'
  };

  function navigate() {
    const hash = location.hash.slice(1) || 'home';

    if (projects[hash]) {
      // Show project view
      homeView.style.display = 'none';
      projectView.style.display = 'block';

      // Create iframe if not already present (per D-03)
      if (!projectView.querySelector('iframe')) {
        const iframe = document.createElement('iframe');
        iframe.src = projects[hash];
        iframe.title = hash.charAt(0).toUpperCase() + hash.slice(1);
        projectView.appendChild(iframe);
      }
    } else {
      // Show home view
      homeView.style.display = '';
      projectView.style.display = 'none';

      // Remove iframe to free resources
      const iframe = projectView.querySelector('iframe');
      if (iframe) iframe.remove();
    }
  }

  // Card click handlers
  document.querySelectorAll('.card[data-project]').forEach(function(card) {
    card.addEventListener('click', function() {
      location.hash = '#' + card.dataset.project;
    });
    card.addEventListener('keydown', function(e) {
      if (e.key === 'Enter' || e.key === ' ') {
        e.preventDefault();
        location.hash = '#' + card.dataset.project;
      }
    });
  });

  // Hash change listener (per D-05 — back button support)
  window.addEventListener('hashchange', navigate);

  // Initial route on page load (per D-06 — direct link support)
  navigate();
})();
</script>
```

This script:
- Maps project names to iframe src paths
- Shows/hides home-view and project-view sections
- Creates iframe on-demand (D-03), removes on return home
- Handles card clicks (converting from href navigation to hash navigation)
- Supports keyboard activation (Enter/Space) for accessibility
- Listens to hashchange for back button support (D-05)
- Calls navigate() on load for direct link support (D-06)

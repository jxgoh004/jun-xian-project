               target_size: int = _DEFAULT_TARGET_SIZE) -> bytes:
        """Render a 60-bar candlestick PNG to bytes, resized to target_size x target_size.

        Args:

                exactly 60 rows (parity with detector._LOOKBACK).

            target_size: final square pixel dimension (default 640 per D-04).

            PNG bytes of the resized RGB image.

        Raises:
            ValueError: if len(df) != 60 or required columns missing.
        """
        if len(df) != _BAR_COUNT:
            raise ValueError(f"render() requires {_BAR_COUNT} bars; got {len(df)}")
        required = {"Open", "High", "Low", "Close"}
        missing = required - set(df.columns)
        if missing:
            raise ValueError(f"render() requires columns {required}; missing: {missing}")

        mpf_style = mpf.make_mpf_style(
            base_mpf_style=style.base_style,
            facecolor=style.facecolor,
            rc={"font.family": _FONT_FAMILY},
        )

        mpf.plot(

            type="candle",
            style=mpf_style,

            axisoff=True,
            savefig=dict(
                fname=buf,
                dpi=style.dpi,
                bbox_inches="tight",
                pad_inches=0,
            ),
        )
        plt.close("all")  # critical: prevents memory growth across thousands of renders

        buf.seek(0)

    def main(argv) -> int:
        """CLI entry: render(<csv>, <style_name>, <out_png>). Returns exit code."""
        if len(argv) < 4:
            print(
                "Usage: python scripts/pattern_scanner/renderer.py "
                "<CSV_PATH> <STYLE_NAME> <OUT_PNG>",

    All tests are pure-unit and run offline using the `synthetic_ohlc` fixture
    from tests/conftest.py. NO @pytest.mark.network markers.
    """
    from __future__ import annotations

    import io
    from typing import List, Tuple

    import numpy as np
    import pandas as pd
    import pytest
    from PIL import Image

    from scripts.pattern_scanner.renderer import (
        RenderStyle,
        STYLES,
        compute_bbox_normalized,
        render,
    )


    # ── Helpers ─────────────────────────────────────────────────────────────────
    def _flat_60(synthetic_ohlc) -> pd.DataFrame:
        """60-bar OHLC frame with monotone slow drift — deterministic, valid for mpf."""
        rng = np.random.default_rng(0)
        rows: List[Tuple[float, float, float, float]] = []
        base = 100.0
        for _ in range(60):
            o = base + rng.uniform(-0.5, 0.5)
            h = max(o, base) + rng.uniform(0.1, 1.0)
            l = min(o, base) - rng.uniform(0.1, 1.0)
            c = base + rng.uniform(-0.5, 0.5)
            rows.append((float(o), float(h), float(l), float(c)))
            base = c
        return synthetic_ohlc(rows)


    # ── Tests ───────────────────────────────────────────────────────────────────
    def test_render_returns_640x640_png(synthetic_ohlc):
        df = _flat_60(synthetic_ohlc)
        png_bytes = render(df, STYLES[0])
        img = Image.open(io.BytesIO(png_bytes))
        assert img.size == (640, 640)
        assert img.mode == "RGB"
        # PNG signature
        assert png_bytes[:8] == b"\x89PNG\r\n\x1a\n"


    def test_render_deterministic_same_style(synthetic_ohlc):
        df = _flat_60(synthetic_ohlc)
        a = render(df, STYLES[0])
        b = render(df, STYLES[0])
        assert a == b, "render must be deterministic on the same machine + style"


    def test_render_styles_differ(synthetic_ohlc):
        df = _flat_60(synthetic_ohlc)
        outs = [render(df, s) for s in STYLES]
        # All three must be byte-distinct.
        assert len(set(outs)) == 3, "STYLES must produce visually distinct PNGs"

        df = _flat_60(synthetic_ohlc).iloc[:50]   # 50 bars, not 60
        with pytest.raises(ValueError, match="60 bars"):

       @pytest.mark.network
       @pytest.mark.parametrize("ticker, expected_date, _t, _s", KNOWN_SETUPS)
       def test_no_detection_on_adjacent_bars(ticker, expected_date, _t, _s):
           df = _fetch(ticker)
           detections = detect(df, ticker)
           # Locate the trading-day index positions for expected_date and its neighbours
           idx = df.index
           pos = idx.get_indexer([pd.Timestamp(expected_date)])[0]
           assert pos != -1, f"{expected_date} not in {ticker} index"
           prev_date = idx[pos - 1].strftime("%Y-%m-%d")
           next_date = idx[pos + 1].strftime("%Y-%m-%d") if pos + 1 < len(idx) else None
           detection_dates = {d.confirmation_date for d in detections}
           assert prev_date not in detection_dates, f"{ticker}: spurious detection on {prev_date} (day before approved {expected_date})"
           if next_date is not None:
               assert next_date not in detection_dates, f"{ticker}: spurious detection on {next_date} (day after approved {expected_date})"

       def test_no_lookahead_truncation_invariance():
           df = _fetch("AAPL")
           full = detect(df, "AAPL")
           # Cut off at 30 bars from the end; detections whose entire 5-bar window completed by the cutoff must be reproduced exactly.
           cutoff_pos = len(df) - 30
           cutoff_date = df.index[cutoff_pos]
           truncated_df = df.iloc[: cutoff_pos + 1]
           truncated = detect(truncated_df, "AAPL")
           # Every detection in `truncated` must appear identically in `full`
           full_keys = {(d.confirmation_date, d.confirmation_type, d.is_spring) for d in full}
           for d in truncated:
               assert (d.confirmation_date, d.confirmation_type, d.is_spring) in full_keys, \
                   f"Truncated frame produced detection not present in full: {d}"
           # Every full detection whose confirmation_idx <= cutoff_pos must appear in `truncated`
           full_in_window = [d for d in full if d.confirmation_bar_index <= cutoff_pos]
           truncated_keys = {(d.confirmation_date, d.confirmation_type, d.is_spring) for d in truncated}
           for d in full_in_window:

                   f"Full detection {d.confirmation_date} missing from truncated frame (look-ahead leak suspected)"
